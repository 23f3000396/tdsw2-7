import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Email: 23f3000396@ds.study.iitm.ac.in

# 1. Load Dataset (Simulated for this analysis)
data = {
    'employee_id': [f'EMP{i:03d}' for i in range(1, 101)],
    'department': ['HR']*15 + ['Finance']*20 + ['R&D']*15 + ['IT']*25 + ['Marketing']*25,
    'region': ['Asia Pacific', 'Europe', 'Middle East', 'Latin America', 'North America'] * 20
}
df = pd.DataFrame(data)

# 2. Calculate Frequency for "Marketing"
marketing_count = df[df['department'] == 'Marketing'].shape[0]
print(f"Frequency count for Marketing department: {marketing_count}")

# 3. Create Histogram
plt.figure(figsize=(10, 6))
sns.histplot(data=df, x='department', shrink=0.8, color='skyblue')
plt.title('Distribution of Employees by Department')
plt.xlabel('Department')
plt.ylabel('Frequency')
plt.grid(axis='y', alpha=0.3)
plt.show()
